<!-- Includi il CSS di Leaflet -->
<link
rel="stylesheet"

/>
<style>

}

<body>
<div id="map"></div>
<!-- Includi la libreria Leaflet -->
<script src="https://unpkg.com/leaflet/dist/leaflet.js"></script>
<script>
// Crea la mappa centrata su una posizione approssimativa delle Azzorre
var map = L.map("map").setView([38.0, -27.5], 7);
// Aggiungi il layer delle mappe da OpenStreetMap
L.tileLayer("https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png", {
maxZoom: 18,
attribution: "© OpenStreetMap",
}).addTo(map);
// Array di oggetti contenenti le località e le rispettive coordinate
var locations = [
{ name: "Algar do Carvão (Terceira)", lat: 38.7136, lon: -27.2150 },
{ name: "Gruta do Natal (Terceira)", lat: 38.7111, lon: -27.2583 },
{ name: "Monte Pico (Pico)", lat: 38.4691, lon: -28.4055 },
{ name: "Vigneti del Pico (Pico)", lat: 38.5314, lon: -28.5250 },
{ name: "Poço da Ribeira do Ferreiro (Flores)", lat: 39.4550, lon: -31.2333 },
{ name: "Lagoa das Sete Cidades (São Miguel)", lat: 37.8736, lon: -25.7950 },
{ name: "Furnas (São Miguel)", lat: 37.7711, lon: -25.3125 },
{ name: "Piantagione di tè Gorreana (São Miguel)", lat: 37.8167, lon: -25.3500 },
{ name: "Piscine naturali di Biscoitos (Terceira)", lat: 38.7833, lon: -27.2333 },
{ name: "Ilhéu de Vila Franca do Campo (São Miguel)", lat: 37.6986, lon: -25.4386 }
];
// Aggiungi un marker per ogni località
locations.forEach(function (loc) {
L.marker([loc.lat, loc.lon])
.addTo(map)
.bindPopup(
"<b>" +
loc.name +
"</b><br>Lat: " +
loc.lat +
", Lon: " +
loc.lon
);
});
</script>
</body>
